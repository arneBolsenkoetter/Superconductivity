\newcommand{\degree}{^\circ}
\newcommand{\strike}[2][\relax]{%
  \begingroup
    \ifx#1\relax
      % No color given → copy current text color into a temp color name
      \colorlet{stcCurrent}{.}%
      \setstcolor{stcCurrent}%
    \else
      % Color given → use that for the strike line only
      \setstcolor{#1}%
    \fi
    \st{#2}%
  \endgroup
}

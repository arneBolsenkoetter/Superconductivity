    \textbf{Executed:}          & #1 \\
    \textbf{Submitted:}         & #2 \\
    \IfValueT{#4}{\textbf{Second Submission:} & #4 \\}
    \IfValueT{#5}{\textbf{Third Submission:}  & #5 \\}

  \end{tabular}%
  \par
  \vspace{\baselineskip}
  \twocolumngrid
  \endgroup
}

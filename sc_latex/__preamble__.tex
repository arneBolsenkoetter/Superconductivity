\newcommand{\strike}[2][\relax]{%
  \begingroup
    \ifx#1\relax
      % No color given → copy current text color into a temp color name
      \colorlet{stcCurrent}{.}%
      \setstcolor{stcCurrent}%
    \else
      % Color given → use that for the strike line only
      \setstcolor{#1}%
    \fi
    \st{#2}%
  \endgroup
}

% usage: \metainfo{<date executed>}{<date submitted>}{<name supervisior>}[<optional second submission date>][<optional third submission date>]
\NewDocumentCommand{\metainfo}{ m m m o o }{%
  \begingroup
  \onecolumngrid
  \vspace{-0.5\baselineskip}
  \noindent
  \begin{tabular}{@{}ll@{}}
    \textbf{Executed:}          & #1 \\
    \textbf{Submitted:}         & #2 \\
    \IfValueT{#4}{\textbf{Second Submission:} & #4 \\}
    \IfValueT{#5}{\textbf{Third Submission:}  & #5 \\}
    \textbf{Supervisor:}        & #3 \\
  \end{tabular}%
  \par
  \vspace{\baselineskip}
  \twocolumngrid
  \endgroup
}
